import pygame import math # Initialize pygame pygame.init() # Screen dimensions WIDTH, HEIGHT = 800, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption('First-Person Car Driving Simulator') # Define colors WHITE = (255, 255, 255) BLACK = (0, 0, 0) RED = (255, 0, 0) # Car class class Car: def __init__(self, x, y, angle): self.x = x self.y = y self.angle = angle # Angle in degrees self.speed = 0 self.image = pygame.Surface((50, 30)) self.image.fill(RED) self.original_image = self.image self.rect = self.image.get_rect(center=(self.x, self.y)) def move(self): # Move car forward or backward rad = math.radians(self.angle) self.x += self.speed * math.cos(rad) self.y -= self.speed * math.sin(rad) self.rect.center = (self.x, self.y) def turn(self, direction): # Turn car left or right self.angle += direction self.angle = self.angle % 360 self.image = pygame.transform.rotate(self.original_image, -self.angle) self.rect = self.image.get_rect(center=self.rect.center) def draw(self, screen): screen.blit(self.image, self.rect) # Initialize car car = Car(WIDTH // 2, HEIGHT // 2, 0) # Game loop running = True clock = pygame.time.Clock() # Game control variables car_speed = 0 turn_speed = 5 # Degrees per key press while running: screen.fill(WHITE) # Event handling for event in pygame.event.get(): if event.type == pygame.QUIT: running = False # Key presses keys = pygame.key.get_pressed() if keys[pygame.K_UP]: car_speed = 5 elif keys[pygame.K_DOWN]: car_speed = -5 else: car_speed = 0 if keys[pygame.K_LEFT]: car.turn(turn_speed) elif keys[pygame.K_RIGHT]: car.turn(-turn_speed) # Update car's position and angle car.speed = car_speed car.move() # Draw the car car.draw(screen) # Update the screen pygame.display.flip() # Limit frames per second clock.tick(60) # Quit pygame pygame.quit()
